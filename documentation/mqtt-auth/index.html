       .port(8555)
       .websocketPort(8999)
       .options(channelOptionMap -> {})
       .ssl(false)
       .reactivePasswordAuth((U,P)->true) //认证注入方法

       .isWebsocket(true)
       .wiretap(false)

       .build()
